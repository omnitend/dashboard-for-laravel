# DXForm

Form object from defineForm

## Live Examples

<ComponentExample
  client:load
  code={`<script setup>
import { DXForm, defineForm } from '@omnitend/dashboard-for-laravel';

const contactForm = defineForm([
  {
    key: 'name',
    label: 'Name',
    type: 'text',
    required: true,
    placeholder: 'Enter your name',
    default: '',
  },
  {
    key: 'email',
    label: 'Email',
    type: 'email',
    required: true,
    placeholder: 'your@email.com',
    default: '',
  },
  {
    key: 'message',
    label: 'Message',
    type: 'textarea',
    required: true,
    placeholder: 'Your message here...',
    default: '',
  },
] as const);

const handleSubmit = () => {
  alert(\`Form submitted! Name: \${contactForm.form.data.name}\`);
};
</script>

<template>
  <DXForm
    :form="contactForm"
    submit-text="Send Message"
    @submit="handleSubmit"
  />
</template>`}
  example="DXFormExample"
/>
